import {
  AfterContentInit,
  Component,
  ContentChildren,
  QueryList,
} from '@angular/core';
import { TabComponent } from '../tab/tab.component';

@Component({
  selector: 'app-tab-list',
  templateUrl: './tab-list.component.html',
  styleUrls: ['./tab-list.component.scss'],
})
export class TabListComponent implements AfterContentInit {
  @ContentChildren(TabComponent) tabList!: QueryList<TabComponent>;

  constructor() {}

  ngAfterContentInit(): void {
    if (this.tabList.some(tab => !tab.isActive)) {
      this.selectTab(this.tabList.first);
    }
  }

  selectTab(tab: TabComponent): void {
    this.tabList.forEach(tab => (tab.isActive = false));
    tab.isActive = true;
  }
}

<ul class="flex">
  <li *ngFor="let tab of tabList; let index = index">
    <button
      class="px-4 py-2 text-sm bg-yellow-700 text-white active"
      [ngClass]="{
        'rounded-tl-lg': index === 0,
        'rounded-tr-lg': index === tabList.length - 1,
        'bg-yellow-900': tab.isActive
      }"
      (click)="selectTab(tab)">
      {{ tab.title }}
    </button>
  </li>
</ul>
